class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int length=0;
        int maxLenght=0;
        for(int i:nums){
            if(i!=1){
                length=0;
            }else{
                length=length+1;
            }
            if(length>maxLenght){
                maxLenght=length;
            }
            
        }
        return maxLenght;
        
    }
}

class Solution {
    public int findNumbers(int[] nums) {
        int count=0;
        for(int num:nums){
            if(numDigit(num)%2==0){count++;}
        }
        return count;
        
    }
    
    int numDigit(int num){
        int rem=num;
        int count=0;
        while(rem!=0){
            rem=rem/10;
            count++;
        }
        return count;
    }
}
